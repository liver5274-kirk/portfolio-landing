實戰一：場景瀏覽器 Extension

在 Extension 中操作 USD 場景並建立自訂 UI。

extension.toml

[package]
version = "1.0.0"
title = "Scene Browser"
category = "example"

[dependencies]
"omni.usd" = {}

[python]
modules = ["omni.example.scene_browser"]

omni/example/scene_browser/__init__.py

import omni.ext
import omni.ui as ui
import omni.usd
from pxr import Usd, UsdGeom


class SceneBrowserExtension(omni.ext.IExt):

    def on_startup(self, ext_id):
        self._window = ui.Window("Scene Browser", width=350, height=500)
        with self._window.frame:
            with ui.VStack(spacing=5):
                # ── 按鈕列 ──
                with ui.HStack(height=30):
                    ui.Button("+ Sphere", clicked_fn=self._add_sphere)
                    ui.Button("+ Cube", clicked_fn=self._add_cube)
                    ui.Button("Refresh", clicked_fn=self._refresh)

                # ── 場景樹 ──
                self._tree = ui.TreeView(
                    ui.SimpleStringModel(""),
                    root_visible=False,
                    header_visible=False
                )
                self._tree.set_selection_changed_fn(self._on_selection_changed)

        self._refresh()

    def on_shutdown(self):
        if self._window:
            self._window.destroy()

    # ── 場景操作 ────────────────────────

    def _get_stage(self):
        return omni.usd.get_context().get_stage()

    def _add_sphere(self):
        stage = self._get_stage()
        if not stage:
            return
        path = omni.usd.get_stage_next_free_path(stage, "/World/Sphere", False)
        omni.kit.commands.execute("CreatePrim",
            prim_type="Sphere", prim_path=path)
        self._refresh()

    def _add_cube(self):
        stage = self._get_stage()
        if not stage:
            return
        path = omni.usd.get_stage_next_free_path(stage, "/World/Cube", False)
        omni.kit.commands.execute("CreatePrim",
            prim_type="Cube", prim_path=path)
        self._refresh()

    def _refresh(self):
        stage = self._get_stage()
        if not stage:
            return

        model = self._tree.model
        model.clear()

        def add_children(parent_item, prim):
            for child in prim.GetChildren():
                name = child.GetName()
                type_name = child.GetTypeName()
                item = ui.SimpleStringModel(f"{name}  [{type_name}]")
                parent_item.add_child(item)
                add_children(item, child)

        root = ui.SimpleStringModel("Stage")
        model.add_child(root)
        add_children(root, stage.GetPseudoRoot())

    def _on_selection_changed(self, items):
        if items:
            print(f"Selected: {items[0].model.get_value_as_string()}")

實戰二：自訂命令系統

註冊自訂命令

import omni.kit.commands
from pxr import UsdGeom, Gf


def create_colored_cube(prim_path: str, color: tuple, size: float = 1.0):
    """自訂命令：建立有顏色的立方體"""
    stage = omni.usd.get_context().get_stage()
    if not stage:
        return

    # 建立 Cube
    cube = UsdGeom.Cube.Define(stage, prim_path)
    cube.GetSizeAttr().Set(size)

    # 設定顏色
    gprim = UsdGeom.Gprim.Get(stage, prim_path)
    gprim.GetDisplayColorAttr().Set([color])

    return prim_path


# 註冊命令
omni.kit.commands.register("CreateColoredCube", create_colored_cube)

# 使用命令
omni.kit.commands.execute("CreateColoredCube",
    prim_path="/World/RedCube",
    color=(1.0, 0.0, 0.0),
    size=2.0
)

實戰三：事件監聽

Kit 使用 Carbonite 事件匯流排：

import omni.kit.app
import carb.events

class EventDemoExtension(omni.ext.IExt):

    def on_startup(self, ext_id):
        # ── 取得事件串流 ──
        self._events = omni.kit.app.get_app().get_message_bus_event_stream()

        # ── 訂閱 Stage 開啟事件 ──
        self._sub_stage_open = omni.usd.get_context().get_stage_event_stream() \
            .create_subscription_to_pop(self._on_stage_event, name="stage")

        # ── 訂閱 Timeline 事件 ──
        self._sub_timeline = self._events.create_subscription_to_pop(
            self._on_timeline_event,
            carb.events.type_from_string("omni.timeline.TimelineEvent")
        )

        # ── 訂閱 Selection 事件 ──
        self._sub_selection = self._events.create_subscription_to_pop(
            self._on_selection_changed,
            carb.events.type_from_string("omni.kit.selection.SelectionChanged")
        )

    def _on_stage_event(self, event):
        if event.type == int(omni.usd.StageEventType.OPENED):
            print("Stage opened!")
        elif event.type == int(omni.usd.StageEventType.CLOSED):
            print("Stage closed!")

    def _on_timeline_event(self, event):
        current_frame = event.payload.get("currentTime", 0)
        is_playing = event.payload.get("isPlaying", False)
        # print(f"Frame: {current_frame}, Playing: {is_playing}")

    def _on_selection_changed(self, event):
        selection = omni.kit.selection.get_selection()
        prims = selection.get_selected_prim_paths()
        print(f"Selected: {prims}")

    def on_shutdown(self):
        self._sub_stage_open = None
        self._sub_timeline = None
        self._sub_selection = None

實戰四：Settings（設定系統）

Extension 可以持久化設定，使用者可在 Preference 中修改。

extension.toml 中宣告設定

[settings]
app/exts/omni.example.my_tool/default_color = [1.0, 0.0, 0.0]
app/exts/omni.example.my_tool/auto_refresh = true
app/exts/omni.example.my_tool/refresh_interval = 5

Python 中讀寫設定

import omni.kit.settings
import carb.settings

class SettingsDemoExtension(omni.ext.IExt):

    def on_startup(self, ext_id):
        # ── 讀取設定（key 對應 extension.toml） ──
        self._settings = carb.settings.acquire_settings_interface()

        default_color = self._settings.get_as_float3(
            "app/exts/omni.example.my_tool/default_color"
        )
        auto_refresh = self._settings.get_as_bool(
            "app/exts/omni.example.my_tool/auto_refresh"
        )
        interval = self._settings.get_as_int(
            "app/exts/omni.example.my_tool/refresh_interval"
        )

        print(f"Color: {default_color}, Auto: {auto_refresh}, Interval: {interval}")

        # ── 寫入設定 ──
        self._settings.set_float3(
            "app/exts/omni.example.my_tool/default_color",
            (0.0, 1.0, 0.0)
        )

        # ── 監聽設定變化 ──
        self._settings_sub = omni.kit.settings.subscribe(
            "app/exts/omni.example.my_tool/", self._on_setting_changed
        )

    def _on_setting_changed(self, key, new_value):
        print(f"Setting changed: {key} = {new_value}")

    def on_shutdown(self):
        self._settings_sub = None

實戰五：USD 屬性編輯面板

完整的 Extension 實作 — 讀取選中 Prim 的屬性並提供編輯介面：

import omni.ext
import omni.ui as ui
import omni.usd
import omni.kit.commands
from pxr import Usd, UsdGeom, Sdf, Gf


class PropertyEditorExtension(omni.ext.IExt):

    def on_startup(self, ext_id):
        self._window = ui.Window("Property Editor", width=350, height=400)
        self._fields = {}  # path → widget

        self._build_ui()

        # 監聽 selection 變化
        self._events = omni.kit.app.get_app().get_message_bus_event_stream()
        self._sel_sub = omni.kit.selection.get_selection().subscribe_to_change_stream(
            self._on_selection_changed
        )

    def _build_ui(self):
        with self._window.frame:
            with ui.ScrollingFrame():
                with ui.VStack(spacing=10, height=0):
                    ui.Label("Property Editor", style={"font_size": 18})

                    self._prim_path_label = ui.Label("No prim selected")

                    # Transform
                    ui.Label("Transform", style={"font_size": 14})
                    with ui.HStack():
                        ui.Label("Translate X:")
                        self._tx = ui.FloatDrag(min=-1000, max=1000)
                        self._tx.model.add_value_changed_fn(
                            lambda m: self._set_transform("translate", 0, m.get_value_as_float())
                        )
                    with ui.HStack():
                        ui.Label("Translate Y:")
                        self._ty = ui.FloatDrag(min=-1000, max=1000)
                    with ui.HStack():
                        ui.Label("Translate Z:")
                        self._tz = ui.FloatDrag(min=-1000, max=1000)

                    # Sphere properties
                    ui.Label("Sphere", style={"font_size": 14})
                    with ui.HStack():
                        ui.Label("Radius:")
                        self._radius = ui.FloatDrag(min=0, max=1000)
                        self._radius.model.add_value_changed_fn(
                            lambda m: self._set_radius(m.get_value_as_float())
                        )

    def _on_selection_changed(self):
        selection = omni.kit.selection.get_selection()
        paths = selection.get_selected_prim_paths()
        if not paths:
            self._prim_path_label.text = "No prim selected"
            return

        path = paths[0]
        self._prim_path_label.text = str(path)

        stage = omni.usd.get_context().get_stage()
        prim = stage.GetPrimAtPath(path)
        if not prim:
            return

        # 讀取並顯示 Transform
        xform = UsdGeom.Xformable(prim)
        ops = xform.GetOrderedXformOps()
        for op in ops:
            if op.GetOpName() == "xformOp:translate":
                val = op.Get()
                self._tx.model.set_value(val[0] if val else 0)
            # ... 可擴充 rotate, scale

        # 讀取並顯示 Radius
        radius_attr = prim.GetAttribute("radius")
        if radius_attr:
            val = radius_attr.Get()
            if val:
                self._radius.model.set_value(val)

    def _set_transform(self, op_name: str, index: int, value: float):
        # 用命令系統做 undoable 編輯
        paths = omni.kit.selection.get_selection().get_selected_prim_paths()
        if not paths:
            return
        omni.kit.commands.execute("ChangeProperty",
            prop_path=Sdf.Path(f"{paths[0]}.xformOp:{op_name}"),
            value=Gf.Vec3d(0, 0, 0),  # 簡化示範
            prev=None
        )

    def _set_radius(self, value: float):
        paths = omni.kit.selection.get_selection().get_selected_prim_paths()
        if not paths:
            return
        omni.kit.commands.execute("ChangeProperty",
            prop_path=Sdf.Path(f"{paths[0]}.radius"),
            value=value,
            prev=None
        )

    def on_shutdown(self):
        self._sel_sub = None
        if self._window:
            self._window.destroy()

Extension 開發除錯技巧

1. Script Editor（內建 Python Console）

Window → Script Editor  (omni.kit.window.script_editor)

直接在執行中的 Kit 測試 Python 程式碼。

2. 熱重載 Extension

Window → Extensions → 找到你的 Extension → 停用 → 啟用

或使用 Script Editor：

import omni.kit.app
mgr = omni.kit.app.get_app().get_extension_manager()
mgr.set_extension_enabled("omni.example.my_tool", False)
mgr.set_extension_enabled("omni.example.my_tool", True)

3. Logging

import carb
carb.log_info("Info message")
carb.log_warn("Warning message")
carb.log_error("Error message")

4. 檢查 Extension 狀態

mgr = omni.kit.app.get_app().get_extension_manager()
exts = mgr.get_extensions()
for ext_id, data in exts.items():
    enabled = mgr.is_extension_enabled(ext_id)
    print(f"{ext_id}: {'✓' if enabled else '✗'}")

Extension 發布

發布到 Local Registry

# 安裝 kit 工具
./kit --empty --exec "omni.kit.pip_archive.pack_extension('path/to/extension')"

發布到 Omniverse Registry

需要在 extension.toml 中加入：

[repository]
url = "https://<your-nucleus-server>/exts/"
