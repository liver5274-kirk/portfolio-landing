一、建立與開啟 Stage

建立新場景

from pxr import Usd, UsdGeom

# 建立空白 Stage
stage = Usd.Stage.CreateNew("myscene.usda")

# 定義 Prim
xform = UsdGeom.Xform.Define(stage, "/World")        # 容器節點
sphere = UsdGeom.Sphere.Define(stage, "/World/Ball")  # 球體
sphere.GetRadiusAttr().Set(50.0)

# 儲存
stage.GetRootLayer().Save()

開啟現有場景

stage = Usd.Stage.Open("myscene.usda")

# 讀取 Prim
prim = stage.GetPrimAtPath("/World/Ball")
radius = prim.GetAttribute("radius").Get()
print(radius)  # 50.0

產生的 .usda 檔案

#usda 1.0

def Xform "World" {
    def Sphere "Ball" {
        double radius = 50
    }
}

二、Geom Schemas（幾何體型別）

Schema Class說明關鍵屬性
UsdGeom.Xform變換容器（群組）xformOp:translate/rotate/scale
UsdGeom.Sphere球體radius, extent
UsdGeom.Cube立方體size, extent
UsdGeom.Cylinder圓柱radius, height, axis
UsdGeom.Cone圓錐radius, height, axis
UsdGeom.Mesh自訂網格points, faceVertexCounts, faceVertexIndices
UsdGeom.Camera攝影機focalLength, fStop

Mesh 建立範例

mesh = UsdGeom.Mesh.Define(stage, "/World/MyMesh")
mesh.CreatePointsAttr([
    (-1, -1, 0), (1, -1, 0), (1, 1, 0), (-1, 1, 0)
])
mesh.CreateFaceVertexCountsAttr([4])           # 一個四邊形
mesh.CreateFaceVertexIndicesAttr([0, 1, 2, 3]) # 頂點順序（逆時針）
mesh.CreateExtentAttr([(-1, -1, 0), (1, 1, 0)])

Xform 操作

from pxr import UsdGeom

xform = UsdGeom.Xformable(stage.GetPrimAtPath("/World"))
# 方法 A：CommonAPI（簡潔）
UsdGeom.XformCommonAPI(xform).SetTranslate((10, 0, 0))
UsdGeom.XformCommonAPI(xform).SetRotate((0, 45, 0))
UsdGeom.XformCommonAPI(xform).SetScale((2, 2, 2))

# 方法 B：Op API（精細控制，支援多個 op 順序）
xform.AddTranslateOp().Set(Gf.Vec3d(10, 0, 0))
xform.AddRotateYOp().Set(45)

四、References（引用外部資產）

核心概念：不複製、只引用，保持資料單一來源。

# 建立一個引用層
refStage = Usd.Stage.CreateNew("MyAssembly.usda")

# 用 Override 建立參考點
chair1 = refStage.OverridePrim("/Chair1")
chair1.GetReferences().AddReference("./chair_model.usda")

chair2 = refStage.OverridePrim("/Chair2")
chair2.GetReferences().AddReference("./chair_model.usda")

# Override 子屬性（不修改原始資產）
overColor = UsdGeom.Sphere.Get(refStage, "/Chair2/cushion")
overColor.GetDisplayColorAttr().Set([(1, 0, 0)])  # Chair2 墊子改紅色

產生的檔案：

over "Chair1" (
    prepend references = @./chair_model.usda@
) {}

over "Chair2" (
    prepend references = @./chair_model.usda@
) {
    over "cushion" {
        color3f[] primvars:displayColor = [(1, 0, 0)]
    }
}

Composed Stage（合成後的完整場景）

# 看合成後的結果（所有引用展開）
print(refStage.ExportToString())

五、Stage 遍歷

簡單遍歷：Traverse()

# 遍歷所有 active prim
for prim in stage.Traverse():
    print(prim.GetPath())  # /World, /World/Ball

# 過濾：只遍歷球體
spheres = [p for p in stage.Traverse() if UsdGeom.Sphere(p)]

# 只遍歷 specific type
spheres = [p for p in stage.Traverse() if p.IsA(UsdGeom.Sphere)]

包含非 active：TraverseAll()

# Traverse() 跳過 active=false 的 prim
# TraverseAll() 包含所有 prim
for prim in stage.TraverseAll():
    print(prim.GetPath())

PreAndPostVisit（前序/後序遍歷）

primIter = iter(Usd.PrimRange.PreAndPostVisit(stage.GetPseudoRoot()))
for prim in primIter:
    if primIter.IsPostVisit():
        print(f"← 離開 {prim.GetPath()}")
    else:
        print(f"→ 進入 {prim.GetPath()}")

輸出範例：

→ 進入 /             # 前序：先父
→ 進入 /World
→ 進入 /World/Ball
← 離開 /World/Ball   # 後序：後子
← 離開 /World
← 離開 /

這在建立 transform hierarchy 時很重要（先處理父節點變換，再處理子節點）。

六、VariantSets（變體切換）

用於 LOD、材質切換、零件替換等場景。

建立 VariantSet

rootPrim = stage.GetPrimAtPath("/World")
vset = rootPrim.GetVariantSets().AddVariantSet("shadingVariant")

# 加入三個變體
vset.AddVariant("red")
vset.AddVariant("green")
vset.AddVariant("blue")

# 在每個變體中編輯
vset.SetVariantSelection("red")
with vset.GetVariantEditContext():
    colorAttr.Set([(1, 0, 0)])

vset.SetVariantSelection("green")
with vset.GetVariantEditContext():
    colorAttr.Set([(0, 1, 0)])

vset.SetVariantSelection("blue")
with vset.GetVariantEditContext():
    colorAttr.Set([(0, 0, 1)])

產生的 .usda：

def Xform "World" (
    variants = {
        string shadingVariant = "green"
    }
    prepend variantSets = "shadingVariant"
) {
    def Sphere "Ball" { ... }

    variantSet "shadingVariant" = {
        "red" {
            over "Ball" {
                color3f[] primvars:displayColor = [(1, 0, 0)]
            }
        }
        "green" {
            over "Ball" {
                color3f[] primvars:displayColor = [(0, 1, 0)]
            }
        }
        "blue" {
            over "Ball" {
                color3f[] primvars:displayColor = [(0, 0, 1)]
            }
        }
    }
}

執行階段切換變體

sessionLayer = stage.GetSessionLayer()
# 用 usdview 或在 session layer 中設定
# 或在 Python 中：
rootPrim.GetVariantSets()["shadingVariant"].SetVariantSelection("red")

七、動畫（Time Samples）

from pxr import Usd, UsdGeom

stage = Usd.Stage.CreateNew("animated.usda")
sphere = UsdGeom.Sphere.Define(stage, "/Ball")

# 設定 time-sampled 動畫
radiusAttr = sphere.GetRadiusAttr()
radiusAttr.Set(1.0, Usd.TimeCode.Default())  # default value
radiusAttr.Set(10.0, Usd.TimeCode(1))        # frame 1
radiusAttr.Set(100.0, Usd.TimeCode(24))      # frame 24
radiusAttr.Set(10.0, Usd.TimeCode(48))       # frame 48

# 讀取特定 frame
print(radiusAttr.Get(Usd.TimeCode(12)))  # 內插值 ≈ 55

# 讀取所有 time samples
for t in radiusAttr.GetTimeSamples():
    val = radiusAttr.Get(t)
    print(f"frame {t}: {val}")

八、完整範例：建立一個組裝場景

from pxr import Usd, UsdGeom, Sdf, Gf, Kind

# 1. 建立資產層（椅子模型）
assetStage = Usd.Stage.CreateNew("chair.usda")
UsdGeom.SetStageUpAxis(assetStage, UsdGeom.Tokens.y)  # Y-up

chair = UsdGeom.Xform.Define(assetStage, "/Chair")
Usd.ModelAPI(chair).SetKind(Kind.Tokens.component)

# 椅面
seat = UsdGeom.Cube.Define(assetStage, "/Chair/Seat")
seat.GetSizeAttr().Set(2.0)
UsdGeom.XformCommonAPI(seat).SetTranslate((0, 1, 0))
UsdGeom.XformCommonAPI(seat).SetScale((2, 0.2, 2))

# 椅背
back = UsdGeom.Cube.Define(assetStage, "/Chair/Back")
back.GetSizeAttr().Set(2.0)
UsdGeom.XformCommonAPI(back).SetTranslate((0, 3, -1))
UsdGeom.XformCommonAPI(back).SetScale((2, 2, 0.2))

assetStage.GetRootLayer().Save()

# 2. 建立組裝層
assemblyStage = Usd.Stage.CreateNew("room.usda")
assemblyStage.SetDefaultPrim(assemblyStage.DefinePrim("/Room"))

# 放置多把椅子
for i, pos in enumerate([(0, 0, 0), (5, 0, 0), (-5, 0, 0)]):
    primPath = f"/Room/Chair{i+1}"
    chair = assemblyStage.OverridePrim(primPath)
    chair.GetReferences().AddReference("./chair.usda", "/Chair")
    UsdGeom.XformCommonAPI(UsdGeom.Xformable(
        assemblyStage.GetPrimAtPath(primPath)
    )).SetTranslate(pos)

assemblyStage.GetRootLayer().Save()
print(assemblyStage.ExportToString())
